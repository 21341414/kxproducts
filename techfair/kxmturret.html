01 — overview
What Was Built
stripped the hackpack turret down to its mechanical base — chassis, barrel assembly, and servo mounts. the original arduino and all custom control circuitry was removed entirely. rewired from scratch with an arduino uno as the new brain, an mpu6050 gyroscope+accelerometer for head-tracking control, and a pressure sensor on the trigger. the turret now mirrors head movement in real time: tilt maps pitch, rotation maps yaw, and squeeze fires.
02 — hardware delta
What Changed
01
MCU Swap
hackpack's proprietary board → arduino uno r3. full i2c + pwm access.
02
Gyro Input Added
mpu6050 wired to sda/scl. provides 6-dof motion data at ~100hz via i2c.
03
Trigger Sensor
pressure sensor on a0. analog read maps squeeze force to fire speed.
04
Full Rewire
all servo signal lines relocated to pins 9, 10, 11 (pwm-capable).
PIN 10
yaw axis
Left / Right
continuous rotation servo — blue wire
driven by gx (gyro x-axis angular velocity). deadzone at ±5000 raw units. writes 105 cw, 75 ccw, 90 to stop. base rotates to follow head turn.
PIN 9
pitch axis
Up / Down
180° positional servo — cyan wire
maps ay (accel y-axis) from –17000..17000 directly to 0..180°. constrained to safe range. barrel angle matches physical head tilt continuously.
PIN 11
fire axis
Fire
continuous rotation servo — yellow wire
idle at 90. pressure above threshold maps 100..1023 → 110..180 firing speed. variable force control — light squeeze = slow rate, full squeeze = max rate.
03 — control logic
Input → Output Map
input axis output type
ay (accel) pitch map → 0–180° positional
gx > 5000 yaw write 105 (cw) velocity
gx < -5000 yaw write 75 (ccw) velocity
gx in deadzone yaw write 90 (stop) idle
pressure > 100 fire map → 110–180 fire
pressure ≤ 100 fire write 90 (stop) idle
04 — tuning values
Constants
pressureThreshold 100 minimum analog reading to engage fire servo. raise to debounce.
gyroDeadzone 5000 gx must exceed ±5000 raw units to trigger yaw rotation.
yawServo.attach pin 10 blue wire — continuous. 90=stop, 75=ccw, 105=cw.
pitchServo.attach pin 9 cyan wire — 180°. mapped linearly to ay reading.
rollServo.attach pin 11 yellow wire — continuous. fire speed maps 110–180.
i2c baud 9600 serial debug rate. mpu6050 on sda/scl (a4/a5 on uno).
05 — wiring
Pin Connections
blue → D10 (PWM) yaw servo / signal
cyan → D9 (PWM) pitch servo / signal
yellow → D11 (PWM) fire servo / signal
red → 5V mpu6050 vcc + servos vcc
black → GND common ground
sda/scl → A4 / A5 mpu6050 i2c bus
pressure → A0 analog signal pin "S"

/*
 * GYRO-TURRET — khxngvng
 * modded hackpack turret base
 * arduino uno + mpu6050 + pressure sensor
 */
#include "Wire.h"
#include "MPU6050.h"
#include "Servo.h"

// ── hardware instances ──────────────────────────
MPU6050 mpu;
Servo   yawServo;    // blue  wire  — continuous rotation
Servo   pitchServo;  // cyan  wire  — 180° positional
Servo   rollServo;   // yellow wire — continuous rotation (fire)

// ── sensor data ─────────────────────────────────
int16_t ax, ay, az, gx, gy, gz;
const int pressurePin = A0;
int pressureVal = 0;

// ── tuning ──────────────────────────────────────
const int pressureThreshold = 100;  // min pressure to fire
const int gyroDeadzone       = 5000; // gx sensitivity threshold

void setup() {
  Serial.begin(9600);
  Wire.begin();

  Serial.println("Initializing MPU6050...");
  mpu.initialize();
  if (!mpu.testConnection()) {
    Serial.println("MPU6050 connection failed!");
  }

  yawServo.attach(10);    // blue  wire
  pitchServo.attach(9);   // cyan  wire
  rollServo.attach(11);   // yellow wire

  yawServo.write(90);    // continuous: 90 = stop
  pitchServo.write(90);  // positional: center
  rollServo.write(90);   // continuous: 90 = stop
}

void loop() {

  // 1. read all sensors ──────────────────────────
  mpu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);
  pressureVal = analogRead(pressurePin);

  // 2. pitch — ay → 0..180° ──────────────────────
  int pitchAngle = map(ay, -17000, 17000, 0, 180);
  pitchAngle   = constrain(pitchAngle, 0, 180);
  pitchServo.write(pitchAngle);

  // 3. yaw — gx deadzone → continuous spin ───────
  if      (gx >  gyroDeadzone) yawServo.write(105);  // cw
  else if (gx < -gyroDeadzone) yawServo.write(75);   // ccw
  else                         yawServo.write(90);   // stop

  // 4. fire — pressure → variable speed ─────────
  if (pressureVal > pressureThreshold) {
    int fireSpeed = map(pressureVal, pressureThreshold, 1023, 110, 180);
    rollServo.write(fireSpeed);
  } else {
    rollServo.write(90);  // idle
  }

  // Serial.print("Pressure: "); Serial.println(pressureVal);
}
